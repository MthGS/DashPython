import streamlit as st
import pandas as pd

# Load the CSV files
male = pd.read_csv("fc25data/male_players.csv")
female = pd.read_csv("fc25data/female_players.csv")

# Set up the Streamlit page configuration
st.set_page_config(
    page_title="Teams",
    page_icon="üõ°Ô∏è",
    layout="wide"
)

# Display logo
st.logo("https://th.bing.com/th/id/OIP.GfYS4X_NrkWCHD5bvAoPegHaAl?rs=1&pid=ImgDetMain")

# Sidebar for team selection
clube = male["Team"].value_counts().index.tolist()  # Get list of team names from male players
team = st.sidebar.selectbox("Clube", clube)

# Filter data for the selected team
team_stats_male = male[male["Team"] == team]
team_stats_female = female[female["Team"] == team]

# Display team title and other details
st.title(f"{team}")
st.markdown(f"**League:** {team_stats_male['League'].iloc[0]}")
st.markdown(f"**Male Players** in {team}:")

# Display the male and female player data side by side in two columns
col1, col2 = st.columns([1, 1], gap="medium")

with col1:
    st.dataframe(team_stats_male.iloc[:, 3:], height=350)  # Show the player data (excluding the first few columns)

with col2:
    if team_stats_female.empty:
        st.markdown("**Nothing found for female players in this team.**")
    else:
        st.markdown(f"**League:** {team_stats_female['League'].iloc[0]}")
        st.markdown(f"**Female Players** in {team}:")
        st.dataframe(team_stats_female.iloc[:, 3:], height=350)  # Show female players' data (excluding first few columns)
